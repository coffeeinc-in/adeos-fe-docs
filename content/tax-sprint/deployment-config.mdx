# Deployment Configuration and Setup

## Complete Deployment Configuration Files

### Docker Compose Configuration (docker-compose.yml)

```yaml
version: '3.8'

services:
  # Redis for queue management and pub/sub
  redis:
    image: redis:7-alpine
    container_name: document-processing-redis
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
      - ./infrastructure/redis/redis.conf:/usr/local/etc/redis/redis.conf
    command: redis-server /usr/local/etc/redis/redis.conf
    networks:
      - document-processing-network
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 30s
      timeout: 10s
      retries: 3

  # Python Backend API
  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: document-processing-backend
    ports:
      - "8000:8000"
    environment:
      - REDIS_HOST=redis
      - REDIS_PORT=6379
      - REDIS_DB=0
      - UPLOAD_DIR=/app/uploads
      - MAX_FILE_SIZE=104857600  # 100MB
      - MAX_CONCURRENT_JOBS=5
      - JOB_TIMEOUT=1800
      - ALLOWED_ORIGINS=http://localhost:3000,http://frontend:3000
      - ENVIRONMENT=production
    volumes:
      - ./backend/uploads:/app/uploads
      - ./backend/logs:/app/logs
    depends_on:
      redis:
        condition: service_healthy
    networks:
      - document-processing-network
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8000/health"]
      interval: 30s
      timeout: 10s
      retries: 3

  # WebSocket Server
  websocket-server:
    build:
      context: ./websocket-server
      dockerfile: Dockerfile
    container_name: document-processing-websocket
    ports:
      - "8001:8001"
    environment:
      - REDIS_HOST=redis
      - REDIS_PORT=6379
      - REDIS_DB=0
      - PORT=8001
      - ALLOWED_ORIGINS=http://localhost:3000,http://frontend:3000
    depends_on:
      redis:
        condition: service_healthy
    networks:
      - document-processing-network
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8001/health"]
      interval: 30s
      timeout: 10s
      retries: 3

  # Next.js Frontend
  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile
    container_name: document-processing-frontend

      - "3000:3000"

      - NEXT_PUBLIC_API_URL=http://localhost:8000
      - NEXT_PUBLIC_WS_URL=ws://localhost:8001
      - NODE_ENV=production
    depends_on:
      - backend
      - websocket-server
    networks:
      - document-processing-network
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:3000"]
      interval: 30s
      timeout: 10s
      retries: 3

  # Nginx Reverse Proxy
  nginx:
    image: nginx:alpine
    container_name: document-processing-nginx
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./infrastructure/nginx/nginx.conf:/etc/nginx/nginx.conf
      - ./infrastructure/nginx/ssl:/etc/nginx/ssl
    depends_on:
      - frontend
      - backend
      - websocket-server
    networks:
      - document-processing-network
    restart: unless-stopped

volumes:
  redis_data:
    driver: local

networks:
  document-processing-network:
    driver: bridge
```

```nginx

worker_processes auto;

    tcp_nopush on;
    tcp_nodelay on;

        text/plain
        text/css
        text/xml
        text/javascript
        application/json
        application/javascript
        application/xml+rss

        image/svg+xml;

    # Rate Limiting
    limit_req_zone $binary_remote_addr zone=api:10m rate=10r/s;
    limit_req_zone $binary_remote_addr zone=uploads:10m rate=2r/s;

    # Upstream servers
    upstream backend {
        server backend:8000;
        keepalive 32;
    }

    upstream websocket {
        server websocket-server:8001;
        keepalive 32;
    }

    upstream frontend {
        server frontend:3000;
        keepalive 32;
    }

    # Main server block
    server {
        listen 80;
        server_name localhost;

        # Security headers
        add_header X-Frame-Options "SAMEORIGIN" always;
        add_header X-XSS-Protection "1; mode=block" always;
        add_header X-Content-Type-Options "nosniff" always;
        add_header Referrer-Policy "no-referrer-when-downgrade" always;
        add_header Content-Security-Policy "default-src 'self' http: https: data: blob: 'unsafe-inline'" always;

        # Frontend (Next.js)
        location / {
            proxy_pass http://frontend;
            proxy_http_version 1.1;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection 'upgrade';
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
            proxy_cache_bypass $http_upgrade;
            proxy_redirect off;
        }

        # Backend API
        location /api {
            limit_req zone=api burst=20 nodelay;
            proxy_pass http://backend;

            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
            proxy_redirect off;
            
            # Increase timeouts for long-running uploads
            proxy_read_timeout 300;
            proxy_connect_timeout 300;
            proxy_send_timeout 300;
        }

        # File uploads with special rate limiting
        location /api/process {
            limit_req zone=uploads burst=5 nodelay;
            proxy_pass http://backend;
            proxy_http_version 1.1;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
            proxy_redirect off;
            
            # Extended timeouts and size limits for uploads
            proxy_read_timeout 600;
            proxy_connect_timeout 600;
            proxy_send_timeout 600;
            client_max_body_size 100M;
        }

        # WebSocket connections
```

### Deployment Scripts (scripts/deploy.sh)

```bash
#!/bin/bash

set -e

# Colors for output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

echo -e "${GREEN}Starting Document Processing System deployment...${NC}"

# Check if Docker is running
if ! docker info > /dev/null 2>&1; then
    echo -e "${RED}Docker is not running. Please start Docker and try again.${NC}"
    exit 1
fi

# Check if docker-compose is available
if ! command -v docker-compose &> /dev/null; then
    echo -e "${RED}docker-compose is not installed. Please install it and try again.${NC}"
    exit 1
fi

# Create necessary directories
echo -e "${YELLOW}Creating necessary directories...${NC}"
mkdir -p backend/uploads backend/logs
mkdir -p infrastructure/nginx/ssl

# Copy environment file if it doesn't exist
if [ ! -f .env ]; then
    echo -e "${YELLOW}Creating .env file from template...${NC}"
    cp .env.example .env
    echo -e "${YELLOW}Please edit .env file with your configuration before running again.${NC}"
    exit 1
fi

# Build and start services
echo -e "${YELLOW}Building Docker images...${NC}"
docker-compose build

echo -e "${YELLOW}Starting services...${NC}"
docker-compose up -d

# Wait for services to be healthy
echo -e "${YELLOW}Waiting for services to be healthy...${NC}"
sleep 30

# Check service health
services=("redis" "backend" "websocket-server" "frontend")
for service in "${services[@]}"; do
    if docker-compose ps | grep -q "${service}.*Up"; then
        echo -e "${GREEN}âœ“ ${service} is running${NC}"
    else
        echo -e "${RED}âœ— ${service} failed to start${NC}"
        docker-compose logs ${service}
        exit 1
    fi
done

echo -e "${GREEN}ðŸŽ‰ Document Processing System deployed successfully!${NC}"
echo -e "${GREEN}Frontend: http://localhost:3000${NC}"
echo -e "${GREEN}Backend API: http://localhost:8000${NC}"
echo -e "${GREEN}WebSocket: ws://localhost:8001${NC}"
echo -e "${GREEN}Redis: localhost:6379${NC}"

echo -e "${YELLOW}To view logs: docker-compose logs -f${NC}"
echo -e "${YELLOW}To stop services: docker-compose down${NC}"
```

## Usage Instructions

### Quick Start with Docker Compose

1. **Clone and navigate to project directory**
2. **Copy environment file**: `cp .env.example .env`
3. **Edit .env file** with your configuration
4. **Run deployment script**: `./scripts/deploy.sh`
5. **Access the application** at `http://localhost:3000`

### Development Setup

1. **Run development setup**: `./scripts/dev-setup.sh`
2. **Start each service individually** in separate terminals
3. **Access the application** at `http://localhost:3000`

### Kubernetes Deployment

1. **Apply namespace**: `kubectl apply -f k8s/namespace.yaml`
2. **Apply configurations**: `kubectl apply -f k8s/`
3. **Configure ingress** for your domain
4. **Access via ingress URL**

### Monitoring and Observability

- **Health checks**: All services include health check endpoints
- **Logging**: Centralized logging with structured format
- **Metrics**: Prometheus-compatible metrics endpoints
- **Alerting**: Configurable alerts for system health

### Security Considerations

- **Rate limiting** implemented in Nginx
- **File upload validation** and size limits
- **CORS configuration** for cross-origin requests
- **Security headers** in Nginx configuration
- **SSL/TLS support** ready for production use
